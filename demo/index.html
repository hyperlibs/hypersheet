🔌 Provider-Driven Grid inline · config · memory · sql · csv · api · database

Every column declares its data provider. The grid resolves options dynamically. Swap a provider type — config, API, database, or static — without changing the grid.

Name Role (inline) Region (memory) Status (inline)
📋 Provider Config
// Provider Registry — define once, reference by name
providers: {
  roles:   { type: 'inline', items: [
    { value: 'admin', label: 'Administrator' },
    { value: 'editor', label: 'Editor' }
  ]},
  regions: { type: 'memory', items: [
    { value: 'na',   label: 'North America' },
    { value: 'eu',   label: 'Europe' }
  ]},
  doctors: { type: 'api', url: '/api/doctors' }
}

// Column references provider by name
{ field: 'role', type: 'dropdown', provider: 'roles' }
{ field: 'region', type: 'dropdown', provider: 'regions' }

// Swap provider — one-line change in registry
{ type: 'csv', source: 'data/regions.csv' }
{ type: 'sql', table: 'doctors', label: 'name', value: 'id' }
{ type: 'api', url: '/api/doctors' }

  required: true,
  email: true,
  maxLength: 100,
  default: '',
  visible: { roles: ['admin', 'editor'] },
  layout: { width: 220 }
}

  default: 'system.today',
  layout: { width: 140, align: 'center' }
